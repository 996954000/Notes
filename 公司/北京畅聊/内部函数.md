private void StartCountDown()
        {
            _tempSchedule = ScheduleClosureHelper.SetClosureSchedule(
                delay: _timeoutSeconds,
                actionClosure: ActionClosure.Create(RefreshingTimeout), // 定时复原标志位
                count: 1 // 仅触发一次
            );
        }

        private void StopCountDown()
        {
            if (_tempSchedule != null)
            {
                ScheduleClosureHelper.ClearClosureSchedule(_tempSchedule);
                _tempSchedule = null;
            }
        }
        
        private void RefreshingTimeout()
        {
            _isRefreshing = false;
        }
