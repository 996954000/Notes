# 原始人目前使用标志位防止网络请求重复或是抢占
计时器来防止标志位错乱，或是未正常复原
private void StartCountDown()
        {
            _tempSchedule = ScheduleClosureHelper.SetClosureSchedule(
                delay: _timeoutSeconds,
                actionClosure: ActionClosure.Create(RefreshingTimeout), // 定时复原标志位
                count: 1 // 仅触发一次
            );
        }

        private void StopCountDown()
        {
            if (_tempSchedule != null)
            {
                ScheduleClosureHelper.ClearClosureSchedule(_tempSchedule);
                _tempSchedule = null;
            }
        }
        
        private void RefreshingTimeout()
        {
            _isRefreshing = false;
        }


// 收包发生错误时重置标志位状态
    private void ResetAllFlagForError()
    {
        _isRefreshing = false;
        _refreshBaseAgainFlag = false;
        _refreshRandAgainFlag = false;
    }

## 1. 代码规范
### 1.1 命名规范
- 变量和函数名应该使用有意义的名称。避免使用单个字母或缩写。
  - 例如，使用“health”代替“hp”。

- 变量名应该以小写字母开头，使用驼峰命名法。
  - 例如，使用“playerHealth”代替“player_health”。

- 函数名应该以动词开头，使用驼峰命名法。
  - 例如，使用“UpdateHealth”代替“healthUpdate”。

- 常量名应该全部大写，使用下划线分隔单词。
  - 例如，使用“MAX_HEALTH”代替“maxHealth”。

- 类名应该使用大写字母开头，使用驼峰命名法。
  - 例如，使用“PlayerController”代替“player_controller”。
